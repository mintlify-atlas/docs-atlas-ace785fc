## Overview

Tabs organize content into multiple panels where only one panel is visible at a time. Built on Base UI's Tabs primitive, they support horizontal and vertical orientations and two visual variants.

## Installation

```bash
npx shadcn@latest add @coss/tabs
```

## Basic usage

```tsx
import { Tabs, TabsList, TabsPanel, TabsTab } from "@/components/ui/tabs"
```

```tsx
<Tabs defaultValue="account">
  <TabsList>
    <TabsTab value="account">Account</TabsTab>
    <TabsTab value="password">Password</TabsTab>
    <TabsTab value="notifications">Notifications</TabsTab>
  </TabsList>
  <TabsPanel value="account">
    <p>Manage your account settings here.</p>
  </TabsPanel>
  <TabsPanel value="password">
    <p>Change your password here.</p>
  </TabsPanel>
  <TabsPanel value="notifications">
    <p>Configure notification preferences here.</p>
  </TabsPanel>
</Tabs>
```

## Sub-components

### Tabs

Root component. Styled wrapper for `Tabs.Root` from Base UI.

<ParamField path="props.defaultValue" type="string">
  The value of the tab that is selected by default (uncontrolled).
</ParamField>

<ParamField path="props.value" type="string">
  The controlled value of the selected tab.
</ParamField>

<ParamField path="props.onValueChange" type="(value: string) => void">
  Called when the selected tab changes.
</ParamField>

<ParamField path="props.orientation" type="'horizontal' | 'vertical'" default="'horizontal'">
  The orientation of the tabs list.
</ParamField>

### TabsList

Container for tab triggers. Styled wrapper for `Tabs.List` from Base UI. Renders an animated indicator for the active tab.

<ParamField path="props.variant" type="'default' | 'underline'" default="'default'">
  `default` renders tabs in a rounded pill container. `underline` renders a bottom border indicator.
</ParamField>

### TabsTab

Individual tab trigger. Styled wrapper for `Tabs.Tab` from Base UI. Also exported as `TabsTrigger`.

<ParamField path="props.value" type="string" required>
  Unique identifier for the tab.
</ParamField>

<ParamField path="props.disabled" type="boolean">
  Prevents interaction with this tab.
</ParamField>

### TabsPanel

Content panel displayed when its matching tab is active. Styled wrapper for `Tabs.Panel` from Base UI. Also exported as `TabsContent`.

<ParamField path="props.value" type="string" required>
  Must match the `value` of a `TabsTab`.
</ParamField>

## Examples

### Underline variant

```tsx
<Tabs defaultValue="overview">
  <TabsList variant="underline">
    <TabsTab value="overview">Overview</TabsTab>
    <TabsTab value="analytics">Analytics</TabsTab>
    <TabsTab value="settings">Settings</TabsTab>
  </TabsList>
  <TabsPanel value="overview">Overview content</TabsPanel>
  <TabsPanel value="analytics">Analytics content</TabsPanel>
  <TabsPanel value="settings">Settings content</TabsPanel>
</Tabs>
```

### Vertical orientation

```tsx
<Tabs defaultValue="general" orientation="vertical">
  <TabsList>
    <TabsTab value="general">General</TabsTab>
    <TabsTab value="security">Security</TabsTab>
    <TabsTab value="billing">Billing</TabsTab>
  </TabsList>
  <TabsPanel value="general">General settings</TabsPanel>
  <TabsPanel value="security">Security settings</TabsPanel>
  <TabsPanel value="billing">Billing settings</TabsPanel>
</Tabs>
```

### Controlled tabs

```tsx
const [tab, setTab] = useState("overview")

<Tabs value={tab} onValueChange={setTab}>
  <TabsList>
    <TabsTab value="overview">Overview</TabsTab>
    <TabsTab value="activity">Activity</TabsTab>
  </TabsList>
  <TabsPanel value="overview">Overview panel</TabsPanel>
  <TabsPanel value="activity">Activity panel</TabsPanel>
</Tabs>
```
