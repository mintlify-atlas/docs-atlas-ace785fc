## Overview

The Date Picker is a composition pattern, not a standalone component. It combines a `Calendar` inside a `Popover` triggered by a `Button`. The button displays the selected date (formatted with `date-fns`) or a placeholder when nothing is selected.

## Installation

Install the required components:

```bash
npx shadcn@latest add @coss/calendar @coss/popover @coss/button
```

Also install `date-fns` for date formatting:

```bash
npm install date-fns
```

## Basic usage

```tsx
"use client"

import { format } from "date-fns"
import { CalendarIcon } from "lucide-react"
import { useState } from "react"

import { Button } from "@/components/ui/button"
import { Calendar } from "@/components/ui/calendar"
import {
  Popover,
  PopoverPopup,
  PopoverTrigger,
} from "@/components/ui/popover"

export function DatePickerDemo() {
  const [date, setDate] = useState<Date>()

  return (
    <Popover>
      <PopoverTrigger
        render={
          <Button
            className="w-64 justify-start text-left font-normal"
            variant="outline"
          />
        }
      >
        <CalendarIcon aria-hidden="true" />
        {date ? format(date, "PPP") : <span>Pick a date</span>}
      </PopoverTrigger>
      <PopoverPopup align="start" className="w-auto p-0">
        <Calendar mode="single" selected={date} onSelect={setDate} />
      </PopoverPopup>
    </Popover>
  )
}
```

## Examples

### Date range picker

```tsx
"use client"

import { format } from "date-fns"
import { CalendarIcon } from "lucide-react"
import { useState } from "react"
import type { DateRange } from "react-day-picker"

import { Button } from "@/components/ui/button"
import { Calendar } from "@/components/ui/calendar"
import { Popover, PopoverPopup, PopoverTrigger } from "@/components/ui/popover"

export function DateRangePicker() {
  const [range, setRange] = useState<DateRange | undefined>()

  const label = range?.from
    ? range.to
      ? `${format(range.from, "LLL dd, y")} – ${format(range.to, "LLL dd, y")}`
      : format(range.from, "LLL dd, y")
    : "Pick a date range"

  return (
    <Popover>
      <PopoverTrigger
        render={
          <Button
            className="w-72 justify-start text-left font-normal"
            variant="outline"
          />
        }
      >
        <CalendarIcon aria-hidden="true" />
        {label}
      </PopoverTrigger>
      <PopoverPopup align="start" className="w-auto p-0">
        <Calendar
          mode="range"
          selected={range}
          onSelect={setRange}
          numberOfMonths={2}
        />
      </PopoverPopup>
    </Popover>
  )
}
```

### With dropdown navigation

Use `captionLayout="dropdown"` for birth date or historical date inputs.

```tsx
"use client"

import { format } from "date-fns"
import { CalendarIcon } from "lucide-react"
import { useState } from "react"

import { Button } from "@/components/ui/button"
import { Calendar } from "@/components/ui/calendar"
import { Popover, PopoverPopup, PopoverTrigger } from "@/components/ui/popover"

export function BirthDatePicker() {
  const [date, setDate] = useState<Date>()

  return (
    <Popover>
      <PopoverTrigger
        render={
          <Button
            className="w-64 justify-start text-left font-normal"
            variant="outline"
          />
        }
      >
        <CalendarIcon aria-hidden="true" />
        {date ? format(date, "PPP") : <span>Date of birth</span>}
      </PopoverTrigger>
      <PopoverPopup align="start" className="w-auto p-0">
        <Calendar
          captionLayout="dropdown"
          fromYear={1920}
          mode="single"
          onSelect={setDate}
          selected={date}
          toYear={new Date().getFullYear()}
        />
      </PopoverPopup>
    </Popover>
  )
}
```

### With presets

```tsx
"use client"

import { addDays, format } from "date-fns"
import { CalendarIcon } from "lucide-react"
import { useState } from "react"

import { Button } from "@/components/ui/button"
import { Calendar } from "@/components/ui/calendar"
import { Popover, PopoverPopup, PopoverTrigger } from "@/components/ui/popover"
import { Select, SelectItem, SelectPopup, SelectTrigger, SelectValue } from "@/components/ui/select"

const presets = [
  { label: "Today", value: "0" },
  { label: "Tomorrow", value: "1" },
  { label: "In 3 days", value: "3" },
  { label: "In a week", value: "7" },
]

export function DatePickerWithPresets() {
  const [date, setDate] = useState<Date>()

  return (
    <Popover>
      <PopoverTrigger
        render={
          <Button
            className="w-64 justify-start text-left font-normal"
            variant="outline"
          />
        }
      >
        <CalendarIcon aria-hidden="true" />
        {date ? format(date, "PPP") : <span>Pick a date</span>}
      </PopoverTrigger>
      <PopoverPopup align="start" className="flex w-auto flex-col space-y-2 p-2">
        <Select
          items={presets}
          onValueChange={(item) => {
            if (item) setDate(addDays(new Date(), parseInt(item.value)))
          }}
        >
          <SelectTrigger>
            <SelectValue placeholder="Select preset" />
          </SelectTrigger>
          <SelectPopup>
            {presets.map((p) => (
              <SelectItem key={p.value} value={p}>{p.label}</SelectItem>
            ))}
          </SelectPopup>
        </Select>
        <Calendar mode="single" selected={date} onSelect={setDate} />
      </PopoverPopup>
    </Popover>
  )
}
```

### Close on select

Control the popover to close when the user picks a date.

```tsx
"use client"

import { format } from "date-fns"
import { CalendarIcon } from "lucide-react"
import { useState } from "react"

import { Button } from "@/components/ui/button"
import { Calendar } from "@/components/ui/calendar"
import { Popover, PopoverPopup, PopoverTrigger } from "@/components/ui/popover"

export function CloseOnSelectDatePicker() {
  const [date, setDate] = useState<Date>()
  const [open, setOpen] = useState(false)

  return (
    <Popover open={open} onOpenChange={setOpen}>
      <PopoverTrigger
        render={
          <Button
            className="w-64 justify-start text-left font-normal"
            variant="outline"
          />
        }
      >
        <CalendarIcon aria-hidden="true" />
        {date ? format(date, "PPP") : <span>Pick a date</span>}
      </PopoverTrigger>
      <PopoverPopup align="start" className="w-auto p-0">
        <Calendar
          mode="single"
          selected={date}
          onSelect={(d) => {
            setDate(d)
            setOpen(false)
          }}
        />
      </PopoverPopup>
    </Popover>
  )
}
```
